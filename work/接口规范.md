# 接口规范

```TS
type CodeType = "000" | "302"

export interface ResponseCommon<T = undefined> {
  traceID: string;
  code: CodeType;
  data: {
    data?: T;
    records?: T;
    record: T;
    page?: Page;
    success: 'true' | 'false';
  };
  message: string;
}

interface Page {
  pageCount: number;
  totalSize: number;
  pageNo: number;
  pageSize: number;
}
```
